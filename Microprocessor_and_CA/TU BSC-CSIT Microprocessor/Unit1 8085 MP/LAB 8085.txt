==================================
8085 division
==================================


;; Question 9/2    9 divident, 2 divisor
;; ANSWER   4 1/2
;; 4 quoitent
;; 1 reminder

;; LOGIC		;; counter C=00h  (for loop i++)
;;9 - 2 = 7   CY=00	c=00
;;7 - 2 = 5   CY=00	c=01
;;5 - 2 = 3   CY=00	c=02
;;3 - 2 = 1   CY=00	c=03	
;;1 - 2 = -1  CY=01

;;-1 + 2 = 1 reminder


;;==========================================

;;Question-------
;;2050h 9 divident, 
;;2051h 2 divisor

;;Answer -----------
;; 2052h 4 quoitent
;; 2053h 1 reminder

;; =========================================

MVI C,255	;; c=ffh load for quoitent

LXI H,2050h	;; memory <-2050	
MOV A,M		;; A<-09
INX H		;; mem = mem + 1 (2051)
MOV B,M		;; B<- 02 (2051)

LOOP: INR C	;; ff+01=00, 00+01=01, 01+01=02, 02+01=03
SUB B		;; A <- A-B
JNC LOOP	;; Check and exit on C=1

ADD B		;; c=1 	A<-A + B  reminder

INX H		;; 2052
MOV M,C		;; quoitent
INX H		;;2053
MOV M,A		;;reminder

HLT
