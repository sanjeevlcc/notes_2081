Below is an example SQL structure for a College Management System 
with two tables and sample data :

Query OK, 1 row affected (0.00 sec)

1. Tables
-------------
Table 1: Students
    This table stores student-related data
    such as student ID, name, course, and enrollment year.


mysql>  CREATE TABLE Students (
        StudentID INT PRIMARY KEY,
        FirstName VARCHAR(50),
        LastName VARCHAR(50),
        Gender VARCHAR(10),
        DateOfBirth DATE,
        EnrollmentYear INT,
        Course VARCHAR(50)
    );



Table 2: Teachers
This table stores teacher-related 
data such as teacher ID, name, department, and salary.


mysql>  CREATE TABLE Teachers (
        TeacherID INT PRIMARY KEY,
        FirstName VARCHAR(50),
        LastName VARCHAR(50),
        Department VARCHAR(50),
        Salary DECIMAL(10, 2),
        HireDate DATE
    );

*********************
Make table 
*********************



CREATE TABLE people (
    person_id        INT PRIMARY KEY,
    role             VARCHAR(20) NOT NULL,   -- 'student','teacher','account','library'
    full_name        VARCHAR(100) NOT NULL,
    gender           CHAR(1),               -- 'M' or 'F'
    department       VARCHAR(50),           -- Management, IT, Health Sciences
    program          VARCHAR(50),           -- BBA, BBA IT, BCA, Medical Lab, MBBS
    semester_or_year TINYINT,               -- 1â€“8 for students, NULL for staff
    email            VARCHAR(100),
    phone            VARCHAR(15),
    join_year        INT                    -- admission year (students) / hire year (staff)
);

SELECT full_name, semester_or_year
FROM people WHERE role = 'student' AND program = 'BBA';




-- 5. Count how many people are in each role (student/teacher/account/library)
SELECT role, COUNT(*) AS total
FROM people
GROUP BY role;




-- 6. List all programs without duplicates
SELECT DISTINCT program
FROM people;




-- 7. Show all MBBS students in year/semester 5 or above
SELECT *
FROM people

  AND program = 'MBBS'
  AND semester_or_year >= 5;

SELECT *
FROM people
WHERE role = 'student'
  AND program = 'Medical Lab'
  AND join_year BETWEEN 2021 AND 2023;
