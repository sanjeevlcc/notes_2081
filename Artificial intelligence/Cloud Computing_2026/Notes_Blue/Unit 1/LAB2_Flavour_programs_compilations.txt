echo "============================================================"
echo " DEMO 8: JAVA WAR APPLICATION USING MAVEN AND TOMCAT"
echo " Concept: Package Java web app and deploy WAR file"
echo "============================================================"

sudo apt install maven -y

cd ~/cloud-demo

if [ ! -d "hellocloudweb" ]; then
    mvn archetype:generate \
    -DgroupId=com.demo \
    -DartifactId=hellocloudweb \
    -DarchetypeArtifactId=maven-archetype-webapp \
    -DinteractiveMode=false

cat > src/main/webapp/index.jsp << 'EOF'
<html>
<head>
    <title>Java WAR Cloud Demo</title>
</head>
<body>
    <h1>Hello World! Welcome to Cloud!</h1>
    <p>This Java WAR application is deployed on Apache Tomcat.</p>
</body>
</html>
EOF

echo "Building WAR file..."
mvn package

echo ""
echo "Deploying WAR file to Tomcat..."

sudo systemctl restart tomcat10

echo ""
echo "Testing WAR application:"
sleep 5

echo ""
echo "Open exposed port 8080 and visit /hellocloudweb/ in browser."

# DEMO 9: PYTHON FLASK WEB APPLICATION
# ============================================================

echo ""
echo "============================================================"
echo " DEMO 9: PYTHON FLASK WEB APPLICATION"
echo " Concept: Simple Python PaaS-style web app"
echo "============================================================"

cd ~/cloud-demo

sudo apt install python3 python3-pip -y

pip3 install flask --break-system-packages 2>/dev/null || pip3 install flask

cat > app.py << 'EOF'
from flask import Flask

app = Flask(__name__)

@app.route("/")
def home():
    return """
    <h1>Hello World! Welcome to Cloud!</h1>
    <p>This Python Flask app is running in a cloud-like environment.</p>
    <p>Cloud Concept: PaaS style application deployment.</p>
    """

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=5000)
EOF

echo ""
echo "Starting Flask app in background on port 5000..."
nohup python3 app.py > flask.log 2>&1 &

sleep 3

echo ""
echo "Testing Flask app:"
curl localhost:5000

echo ""
echo "Open exposed port 5000 in KillerCoda to view Flask app."

const server = http.createServer((req, res) => {
    res.writeHead(200, {"Content-Type": "text/html"});
    res.end(`
        <h1>Hello World! Welcome to Cloud!</h1>
        <p>This Node.js app is running in a cloud-like Ubuntu environment.</p>
        <p>Cloud Concept: PaaS style web application.</p>
    `);
});

server.listen(3000, "0.0.0.0", () => {
    console.log("Server running on port 3000");
});
EOF

echo ""

nohup node server.js > node.log 2>&1 &

echo "Testing Node.js app:"

============================================================
Cloud Service Models
============================================================

1. SaaS - Software as a Service
   - User only uses ready-made software.
   - No need to manage server, OS, database, or code.
   - Examples: Gmail, Google Drive, Canva, Zoom.

2. PaaS - Platform as a Service
   - Developer uploads code.
   - Platform manages server, runtime, and environment.
   - Examples: Heroku, Google App Engine, Azure App Service.

3. IaaS - Infrastructure as a Service
   - User gets virtual server, storage, and network.
   - User manages OS, runtime, and application.
   - Examples: AWS EC2, Azure VM, Google Compute Engine.

============================================================
Trainer Message:
Hello World! Welcome to Cloud!
============================================================
EOF
